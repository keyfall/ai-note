语料库自动化demo
```
import os
import re
import jieba.posseg as pseg  # 使用 jieba 的词性标注模块
import pandas as pd
import requests

class CorpusBuilder:
    def __init__(self, output_dir="corpus"):
        # 初始化输出目录
        self.output_dir = output_dir
        if not os.path.exists(output_dir):
            os.makedirs(output_dir)

    def collect_data(self, url):
        """从指定 URL 采集数据"""
        try:
            response = requests.get(url)
            response.raise_for_status()
            return response.text
        except requests.RequestException as e:
            print(f"数据采集失败: {e}")
            return None

    def clean_text(self, text):
        """对采集的文本进行校对和降噪"""
        # 去除 HTML 标签
        text = re.sub(r'<[^>]+>', '', text)
        # 去除特殊字符和多余空格
        text = re.sub(r'[^\w\s]', '', text)
        text = re.sub(r'\s+', ' ', text).strip()
        return text

    def tokenize_and_tag(self, text):
        """对文本进行分词和词性标注"""
        words_with_tags = pseg.cut(text)  # 使用 jieba 的词性标注功能
        tagged_words = [(word, flag) for word, flag in words_with_tags]
        return tagged_words

    def persist_corpus(self, data, filename="corpus.csv"):
        """将数据持久化到 CSV 文件"""
        filepath = os.path.join(self.output_dir, filename)
        df = pd.DataFrame(data, columns=["Word", "Tag"])
        df.to_csv(filepath, index=False, encoding="utf-8")
        print(f"语料已保存到 {filepath}")

    def build(self, url):
        """完整流程：采集 -> 校对降噪 -> 分词标注 -> 持久化"""
        print("开始采集数据...")
        raw_text = self.collect_data(url)
        if not raw_text:
            print("采集失败，退出构建流程。")
            return

        print("正在进行校对和降噪...")
        cleaned_text = self.clean_text(raw_text)

        print("正在进行分词和词性标注...")
        tagged_words = self.tokenize_and_tag(cleaned_text)

        print("正在持久化语料...")
        self.persist_corpus(tagged_words)

if __name__ == "__main__":
    # 示例 URL（请替换为实际的数据源）
    sample_url = "https://example.com/sample-text"
    
    # 创建语料库构建器实例
    builder = CorpusBuilder()

    # 执行全流程
    builder.build(sample_url)
```
